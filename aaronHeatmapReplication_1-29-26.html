#!/usr/bin/env bash
set -euo pipefail
export MPLBACKEND=Agg

# =========================
# INPUT BASE DIRECTORY
# =========================
BASEDIR="/Users/anna_dorothea_senft/Downloads/Aaron_heatmap_replication"

# =========================
# INPUT FILES
# =========================
REGIONS_BED="${BASEDIR}/mm10_genes_Zfp777_closestTSS_uniqueGenes.bed"
BIGWIG_TSV="${BASEDIR}/tracks_mm10.tsv"   # label <TAB> /path/to/bw [<TAB> colormap]

# =========================
# PARAMETERS
# =========================
UPSTREAM=2000
DOWNSTREAM=2000
BIN_SIZE=25
THREADS=4

# =========================
# OUTPUTS
# =========================

mkdir -p "${OUTDIR}"

OUTDIR="${BASEDIR}/deeptools_TSS"
mkdir -p "${OUTDIR}"

BW_LIST="${OUTDIR}/bigwigs.list.txt"
LABELS_LIST="${OUTDIR}/labels.list.txt"
CMAP_LIST="${OUTDIR}/cmaps.list.txt"

# Reset lists
: > "${BW_LIST}"
: > "${LABELS_LIST}"
: > "${CMAP_LIST}"

# Robust parse: bigwig = token ending in .bw ; colormap = last token after it (if any) ; label = col1
awk '
NR==1 {next}          # skip header
NF==0 {next}          # skip blank lines
{
  gsub("\r","")
  bw=""; bw_i=0
  for (i=1; i<=NF; i++) if (tolower($i) ~ /\.(bw|bigwig)$/) { bw=$i; bw_i=i }
  if (bw=="") { print "PARSE_ERROR_NO_BW\t" $0 > "/dev/stderr"; next }

  cmap="RdYlBu"
  if (NF > bw_i) cmap=$NF
  if (cmap ~ /\.bw$/) cmap="RdYlBu"

  # If your label is clean in col1, keep it simple:
  label=$1

  print bw    >> bw_out
  print label >> lab_out
  print cmap  >> cmap_out
}
' bw_out="${BW_LIST}" lab_out="${LABELS_LIST}" cmap_out="${CMAP_LIST}" "${BIGWIG_TSV}"


SORTED_REGIONS="${OUTDIR}/regions_sorted_by_track1.bed"
MATRIX_ALL="${OUTDIR}/matrix_TSS_ALL.gz"
OUTPDF_MULTI="${OUTDIR}/heatmap_TSS_multipanel_ALLtracks.pdf"

# =========================
# PARSE TSV (ASSUMES HEADER ROW)
#   - col 1: label
#   - col 2: bigwig path
#   - col 3: optional colormap name
# =========================
# awk -F'\t' 'NR>1 && NF>=2 {gsub("\r",""); print $2}' "${BIGWIG_TSV}" > "${BW_LIST}"
# awk -F'\t' 'NR>1 && NF>=1 {gsub("\r",""); print $1}' "${BIGWIG_TSV}" > "${LABELS_LIST}"
# PARSE "TSV" that may be space- or tab-delimited:
# col1=label  col2=bigwig  col3=optional colormap
: > "${BW_LIST}"
: > "${LABELS_LIST}"
: > "${CMAP_LIST}"

awk '
NR==1 {next}  # skip header
{
  gsub("\r","")
  bw=""; cmap=""; label=""
  # last field might be a colormap name (e.g. Reds, RdYlBu, PRGn)
  cmap=$NF

  # find the field that ends with .bw (bigWig)
  for (i=1; i<=NF; i++) {
    if (tolower($i) ~ /\.(bw|bigwig)$/) { bw=$i; bw_i=i }
  }

  if (bw=="") {
    print "PARSE_ERROR_NO_BW\t" $0 > "/dev/stderr"
    next
  }

  # label is everything before the .bw field (joined with spaces)
  for (i=1; i<bw_i; i++) {
    label = (label=="" ? $i : label " " $i)
  }

  # if the last token is actually the bw (no colormap column), set default colormap
  if (cmap ~ /\.bw$/) { cmap="RdYlBu" }

  print bw  >> bw_out
  print label >> lab_out
  print cmap >> cmap_out
}
' bw_out="${BW_LIST}" lab_out="${LABELS_LIST}" cmap_out="${CMAP_LIST}" "${BIGWIG_TSV}"

# optional colormap column
awk -F'\t' 'NR>1 {gsub("\r",""); if (NF>=3 && $3!="") print $3; else print "RdYlBu"}' "${BIGWIG_TSV}" > "${CMAP_LIST}"

# =========================
# BUILD ARGUMENT LISTS (macOS bash 3.2 compatible)
# =========================
BIGWIGS=""
while IFS= read -r bw; do
  [[ -z "$bw" ]] && continue
  BIGWIGS="${BIGWIGS} ${bw}"

  LABELS="${LABELS} ${lab}"

while IFS= read -r cm; do
  [[ -z "$cm" ]] && continue
  CMAPS="${CMAPS} ${cm}"
done < "${CMAP_LIST}"

# =========================
# 1) Sort regions once using track 1 (first bigwig in TSV)
# =========================
BW1=$(sed -n '1p' "${BW_LIST}")
LABEL1=$(sed -n '1p' "${LABELS_LIST}")

MATRIX_1="${OUTDIR}/matrix_TSS_track1_for_sort.gz"

computeMatrix reference-point \
  --referencePoint TSS \
  -R "${REGIONS_BED}" \
  -S "${BW1}" \

  --binSize "${BIN_SIZE}" \
  --missingDataAsZero \

  --plotType lines \
  --refPointLabel "TSS" \
  --samplesLabel "${LABEL1}" \
  --averageTypeSummaryPlot mean \
  --sortRegions descend \
  --sortUsing mean \
  --outFileSortedRegions "${SORTED_REGIONS}"

# =========================
# 2) Compute ONE matrix for ALL tracks using the sorted region order
# =========================
computeMatrix reference-point \
  --referencePoint TSS \
  -R "${SORTED_REGIONS}" \
  -S ${BIGWIGS} \
  -b "${UPSTREAM}" -a "${DOWNSTREAM}" \
  --binSize "${BIN_SIZE}" \
  --missingDataAsZero \
  --numberOfProcessors "${THREADS}" \
  -o "${MATRIX_ALL}"

# =========================
# 3) Plot ONCE: multipanel across samples
#   - Multiple --colorMap entries => per-panel colorbars along bottom (student-style)
#   - Autoscaling: do NOT set --zMin/--zMax or --yMin/--yMax
# =========================
# =========================
